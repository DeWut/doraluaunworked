-- Оставляем только необходимые декларации для консоли (как в Chimera)
ffi.cdef[[
    typedef void* (__cdecl* tCreateInterface)(const char*, int*);
    typedef void(__thiscall* execute_cmd_t)(void*, const char*);
    void* GetModuleHandleA(const char* lpModuleName);
    void* GetProcAddress(void* hModule, const char* lpProcName);
]]

local engine_ptr = (function()
    local h_mod = ffi.C.GetModuleHandleA("engine.dll")
    if h_mod == nil then return nil end
    local create_interface = ffi.cast("tCreateInterface", ffi.C.GetProcAddress(h_mod, "CreateInterface"))
    return create_interface("VEngineClient014", nil)
end)()

-- Безопасный оверрайд (Фундаментальная функция для аима и АА)
function utils.safe_override(ref, value)
    if ref and ref.override then
        if value ~= nil then ref:override(value) else ref:override() end
    end
end

-- Упрощенная проверка сетевого канала (как в Chimera)
function utils.is_connected()
    if not engine_ptr then return false end
    local vtbl = ffi.cast("void***", engine_ptr)[0]
    local get_net_info_fn = ffi.cast("void*(__thiscall*)(void*)", vtbl[78])
    return get_net_info_fn(engine_ptr) ~= nil
end

-- Выполнение команд через движок
function utils.console_exec(cmd)
    if not engine_ptr then return end
    local vtbl = ffi.cast("void***", engine_ptr)[0]
    local fn = ffi.cast("execute_cmd_t", vtbl[108])
    fn(engine_ptr, cmd)
end

function utils.safe_ui_find(...)
    local success, result = pcall(ui.find, ...)
    if success and result then return result end
    local args = {...}
    if args[2] == "Anti-Aim" then args[2] = "Anti Aim"
    elseif args[2] == "Anti Aim" then args[2] = "Anti-Aim" end
    local success2, result2 = pcall(ui.find, unpack(args))
    return success2 and result2 or nil
end

function utils.lerp(a, b, t) return a + (b - a) * t end
function utils.color_lerp(c1, c2, t)
    return color(utils.lerp(c1.r, c2.r, t), utils.lerp(c1.g, c2.g, t), utils.lerp(c1.b, c2.b, t), utils.lerp(c1.a, c2.a, t))
end

-- Функция применения пресета
function funcs.apply_preset(preset_name)
    local preset = presets[preset_name]
    if not preset then return false end
    
    local s = preset.settings
    
    -- Вспомогательная функция для безопасной установки значений
    local function safe_set(ui_element, value)
        if ui_element and value ~= nil then
            ui_element:set(value)
        end
    end
    
    -- Применяем настройки фликов
    safe_set(cfg.flick_mode, s.flick_mode)
    safe_set(cfg.flick_speed, s.flick_speed)
    safe_set(cfg.flick_angle, s.flick_angle)
    safe_set(cfg.flick_delay, s.flick_delay)
    safe_set(cfg.flick_inverter_on_flick, s.flick_inverter)
    
    -- Индикаторы
    safe_set(cfg.indicator_name, s.indicator_name)
    safe_set(cfg.indicator_font, s.indicator_font)
    safe_set(cfg.indicator_pos_y, s.indicator_pos_y)
    safe_set(cfg.indicator_mode, s.indicator_mode)
    safe_set(cfg.indicator_extra, s.indicator_extra)
    safe_set(cfg.gradient_speed, s.gradient_speed or 0)
    
    -- Viewmodel
    safe_set(cfg.vm_enable, s.vm_enable)
    safe_set(cfg.vm_fov, s.vm_fov)
    safe_set(cfg.vm_offset_x, s.vm_offset_x)
    safe_set(cfg.vm_offset_y, s.vm_offset_y)
    safe_set(cfg.vm_offset_z, s.vm_offset_z)
    safe_set(cfg.vm_opposite_knife, s.vm_opposite_knife)
    
    if cfg.vm_enable and cfg.vm_enable:get() then 
        if funcs.apply_viewmodel then funcs.apply_viewmodel() end 
    end
    
    -- Console
    safe_set(cfg.console_color_enable, s.console_color_enable)
    if cfg.console_color_enable and cfg.console_color_enable:get() then 
        if funcs.apply_console_color then funcs.apply_console_color() end 
    end
    
    -- Aspect
    safe_set(cfg.aspect_enable, s.aspect_enable)
    safe_set(cfg.aspect_value, s.aspect_value)
    if cfg.aspect_enable and cfg.aspect_enable:get() then 
        if funcs.apply_aspect then funcs.apply_aspect() end 
    end
    
    -- Прочее
    safe_set(cfg.zeus_enable, s.zeus_enable)
    safe_set(cfg.jumpscout_enable, s.jumpscout_enable)
    safe_set(cfg.name_enable, s.name_enable)
    
    -- AA States
    if cfg.aa_state_configs then
        local states = {"Standing", "Moving", "Air", "Crouch"}
        for _, state_name in ipairs(states) do
            local config = cfg.aa_state_configs[state_name]
            local prefix = "aa_" .. state_name:lower() .. "_"
            
            if config then
                safe_set(config.enabled, s[prefix .. "enabled"])
                safe_set(config.pitch, s[prefix .. "pitch"])
                safe_set(config.yaw, s[prefix .. "yaw"])
                safe_set(config.yaw_offset_left, s[prefix .. "offset_left"])
                safe_set(config.yaw_offset_right, s[prefix .. "offset_right"])
                safe_set(config.inverter, s[prefix .. "inverter"])
                safe_set(config.left_limit, s[prefix .. "left_limit"])
                safe_set(config.right_limit, s[prefix .. "right_limit"])
            end
        end
    end
    
    print(string.format("\a#E195FF[Dora.Lua]\a#FFFFFF Пресет '%s' загружен!", preset.name))
    return true
end

-- 11.1 Безопасное получение библиотеки файлов
local function get_files_lib()
    return files or require("neverlose/files")
end

-- 11.2 Получение списка сохранённых конфигов
function funcs.get_saved_configs()
    local configs = {}
    local fl = get_files_lib()
    
    local status, all_files = pcall(fl.find, "nl/dora_config_*.json")
    
    if status and all_files then
        for _, path in ipairs(all_files) do
            local filename = path:match("([^/\\]+)$")
            if filename then
                local match = filename:match("^dora_config_(.+)%.json$")
                if match then table.insert(configs, match) end
            end
        end
    end
    return configs
end

-- 11.3 Экспорт/Импорт JSON (рекурсивный)
function funcs.export_config_json()
    local function serialize(t, output)
        for key, element in pairs(t) do
            if type(element) == "table" then
                output[key] = {}
                serialize(element, output[key])
            elseif type(element) == "userdata" and pcall(function() return element.get end) then
                local success, val = pcall(element.get, element)
                if success then
                    if type(val) == "userdata" and val.r then
                        output[key] = { __type = "color", r = val.r, g = val.g, b = val.b, a = val.a }
                    elseif type(val) ~= "userdata" and type(val) ~= "function" then
                        output[key] = val
                    end
                end
            end
        end
    end
    local config_to_save = {}
    serialize(cfg, config_to_save)
    return json.encode(config_to_save)
end

function funcs.import_config_json(json_str)
    local status, decoded = pcall(json.decode, json_str)
    if not status or type(decoded) ~= "table" then return false end
    local function deserialize(t_cfg, t_data)
        for key, value in pairs(t_data) do
            if type(value) == "table" and not value.__type then
                if t_cfg[key] then deserialize(t_cfg[key], value) end
            elseif t_cfg[key] and t_cfg[key].set then
                if type(value) == "table" and value.__type == "color" then
                    t_cfg[key]:set(color(value.r, value.g, value.b, value.a))
                else
                    t_cfg[key]:set(value)
                end
            end
        end
    end
    deserialize(cfg, decoded)
    return true
end

-- 11.4 Сохранение и загрузка
function funcs.save_config_to_file(name)
    if not name or name == "" then return end
    local encoded = funcs.export_config_json()
    local fl = get_files_lib()
    local path = "nl/dora_config_" .. name .. ".json"
    
    if fl.write(path, encoded) then
        print(string.format("\a#E195FF[Dora.Lua]\a#FFFFFF Конфиг '%s' сохранён!", name))
        funcs.refresh_config_list()
    else
        print("\a#FF0000[Dora.Lua] Ошибка! Создайте папку: /Neverlose/nl/")
    end
end

function funcs.load_config_from_file(name)
    local fl = get_files_lib()
    local path = "nl/dora_config_" .. name .. ".json"
    local content = fl.read(path)
    if content and funcs.import_config_json(content) then
        print(string.format("\a#E195FF[Dora.Lua]\a#FFFFFF Конфиг '%s' загружен!", name))
    end
end

function funcs.refresh_config_list()
    local list = funcs.get_saved_configs()
    table.insert(list, 1, "Select config...")
    if cfg.config_select then cfg.config_select:update(list) end
end

-- Функция определения текущего состояния
function funcs.get_aa_state(lp, weapon, cmd)
    if not lp or not lp:is_alive() then return "Standing" end
    local flags = lp.m_fFlags
    local on_ground = bit.band(flags, 1) == 1
    local duck = lp.m_flDuckAmount > 0.7
    local vel = lp.m_vecVelocity:length2d()
    if not on_ground then return duck and "Crouch" or "Air" end
    if weapon and weapon:get_classid() == 34 then return "InUse" end
    if cfg.refs.aa_slow_walk and (cfg.refs.aa_slow_walk:get_override() or cfg.refs.aa_slow_walk:get()) then return "SlowWalk" end
    if duck then return "Crouch" end
    if vel < 1.1 then return "Standing" end
    return "Moving"
end

-- Применение настроек для текущего состояния (Better AA)
function funcs.apply_aa_state(state_name)
    -- Проверка мастер-переключателя Better AA
    if not cfg.better_aa_enable or not cfg.better_aa_enable:get() then return end
    local cfg_state = cfg.aa_state_configs[state_name]
    if not cfg_state or not cfg_state.enabled:get() then return end
    local p = cfg_state.pitch:get()
    if p ~= "Default" then utils.safe_override(cfg.refs.aa_pitch, p) else utils.safe_override(cfg.refs.aa_pitch) end
    local y = cfg_state.yaw:get()
    if y ~= "Default" then utils.safe_override(cfg.refs.aa_yaw, y) else utils.safe_override(cfg.refs.aa_yaw) end
    local inv = cfg_state.inverter:get()
    local off = inv and cfg_state.yaw_offset_left:get() or cfg_state.yaw_offset_right:get()
    utils.safe_override(cfg.refs.aa_yaw_offset, off)
    local mod = cfg_state.yaw_modifier:get()
    local deg = cfg_state.modifier_degree:get()
    if mod == "Center" then
        rage.antiaim:override_hidden_yaw_offset(deg)
    elseif mod == "Offset" then
        rage.antiaim:override_hidden_yaw_offset(off + deg)
    elseif mod == "Random" then
        rage.antiaim:override_hidden_yaw_offset(off + math.random(-deg, deg))
    end
    utils.safe_override(cfg.refs.aa_body_yaw, cfg_state.body_yaw:get())
    utils.safe_override(cfg.refs.aa_inverter, inv)
    utils.safe_override(cfg.refs.aa_left_limit, cfg_state.left_limit:get())
    utils.safe_override(cfg.refs.aa_right_limit, cfg_state.right_limit:get())
end

state.chat_queue = { queue = {}, last_sent_time = 0 }
function state.chat_queue:add(msg, delay)
    table.insert(self.queue, { text = msg, delay = delay })
    self.last_sent_time = globals.curtime
end
function state.chat_queue:process()
    if #self.queue == 0 then return end
    local now = globals.curtime
    local first = self.queue[1]
    if now - self.last_sent_time >= first.delay then
        utils.console_exec("say " .. first.text)
        table.remove(self.queue, 1)
        self.last_sent_time = now
    end
end
function state.chat_queue:clear() self.queue = {}; self.last_sent_time = 0 end

-- 9.1 Обновление фликов
function funcs.update_flick(cmd, lp)
    -- Проверка мастер-переключателя Flicks
    if not cfg.flick_enable or not cfg.flick_enable:get() then
        state.current_flick_angle = 0
        return 0
    end
    local mode = ui_elements.flick_mode:get()
    if mode == "None" then return 0 end

    local speed = cfg.flick_speed:get()
    local angle = cfg.flick_angle:get()
    local delay = cfg.flick_delay:get()

    state.flick_tick = state.flick_tick + 1

    if not state.flick_active then
        if state.flick_tick % (speed + delay) == 0 then
            state.flick_active = true
            state.last_flick_tick = globals.tickcount
            if mode == "Classic Flick" then
                state.current_flick_angle = angle
            elseif mode == "Fake Flick" then
                if cfg.flick_inverter_on_flick:get() then
                    utils.safe_override(cfg.refs.aa_inverter, not (cfg.refs.aa_inverter:get_override() or cfg.refs.aa_inverter:get()))
                end
                state.current_flick_angle = angle
            elseif mode == "Defensive Flick" then
                if cmd.choke > 0 then
                    state.current_flick_angle = angle
                else
                    state.current_flick_angle = 0
                end
            elseif mode == "Double Flick" then
                state.current_flick_angle = angle * (state.flick_tick % 2 == 0 and 1 or -1)
            elseif mode == "Random Flick" then
                state.current_flick_angle = math.random(-angle, angle)
            end
        end
    else
        if globals.tickcount - state.last_flick_tick >= 1 then
            state.flick_active = false
            state.current_flick_angle = 0
        end
    end

    return state.current_flick_angle
end

-- 9.4 AI PEEK
function funcs.ai_peek_on_createmove(cmd)
    -- Проверяем включён ли мастер-переключатель Advanced Rage и сам AI Peek
    if not cfg.advanced_rage_enable or not cfg.advanced_rage_enable:get() then return end
    if not cfg.ai_peek_enable:get() then return end
    local lp = entity.get_local_player()
    if not lp or not lp:is_alive() then return end
    local weapon = lp:get_player_weapon()
    if not weapon then return end
    local dt = cfg.refs.dt:get_override() or cfg.refs.dt:get()
    local hs = cfg.refs.hs:get_override() or cfg.refs.hs:get()
    if not (dt or hs) then return end
    if not (cfg.refs.peek_assist:get_override() or cfg.refs.peek_assist:get()) then return end
    local weapon_type = nil
    for _, v in pairs(cfg.ai_peek_weapons:get()) do
        if v == weapon:get_weapon_info().weapon_type then weapon_type = v; break end
    end
    if not weapon_type then return end

    if globals.curtime - (state.ai_peek_last_calc or 0) < cfg.ai_peek_rate:get() * 0.01 then return end

    local enemy = entity.get_threat(true)
    if not enemy or enemy:is_dormant() then return end

    local sim_ticks = math.floor(cfg.ai_peek_sim:get() * 0.01 / globals.tickinterval)
    local min_damage = cfg.refs.min_damage:get_override() or cfg.refs.min_damage:get()
    if min_damage < 12 then min_damage = 12 end

    local eye_pos = lp:get_eye_position()
    local enemy_origin = enemy:get_origin()
    local best_angle = nil
    local best_damage = 0
    local best_pos = nil

    for offset = -90, 90, 45 do
        local test_yaw = math.normalize_yaw((eye_pos:to(enemy_origin)):angles().y + offset)
        local test_angles = vector(0, test_yaw, 0)
        local sim_pos, sim_vel = funcs.simulate_movement(lp:get_origin(), test_angles, cmd, sim_ticks)
        local new_eye = sim_pos + vector(0,0, lp:get_view_offset().z)
        local trace = utils.trace_bullet(lp, new_eye, enemy_origin, function(e) return e == enemy end)
        if trace and trace[1] >= min_damage then
            local hitchance = 100
            if hitchance >= cfg.ai_peek_hc:get() then
                if trace[1] > best_damage then
                    best_damage = trace[1]
                    best_angle = test_angles
                    best_pos = sim_pos
                end
            end
        end
    end

    if best_angle then
        state.ai_peek_active = true
        state.ai_peek_target = enemy
        state.ai_peek_angle = best_angle
        state.ai_peek_pos = best_pos
        local move_dir = (best_pos - lp:get_origin()):normalized()
        cmd.move_yaw = move_dir:angles().y
        cmd.forwardmove = 450
        cmd.sidemove = 0
        cmd.block_movement = 1
        if lp:get_origin():dist2d(best_pos) < 20 then
            cmd.in_attack = true
            cmd.view_angles = best_angle
            state.ai_peek_active = false
        end
    else
        state.ai_peek_active = false
    end
    state.ai_peek_last_calc = globals.curtime
end

function funcs.simulate_movement(origin, angles, cmd, steps)
    local vel = vector(0,0,0)
    local pos = origin:clone()
    local dir = vector():angles(angles):vectors()
    for _ = 1, steps do
        vel = vel + dir * 450 * globals.tickinterval
        vel = vel * 0.98
        pos = pos + vel * globals.tickinterval
    end
    return pos, vel
end

-- 9.5 DORMANT AIMBOT
function funcs.dormant_aimbot_on_createmove(cmd)
    if not cfg.advanced_rage_enable or not cfg.advanced_rage_enable:get() then return end
    if not cfg.dormant_enable:get() then return end
    local lp = entity.get_local_player()
    if not lp or not lp:is_alive() then return end
    local weapon = lp:get_player_weapon()
    if not weapon then return end
    local weapon_info = weapon:get_weapon_info()
    if not weapon_info then return end

    local eye_pos = lp:get_eye_position()
    local candidates = {}
    local min_dmg = cfg.dormant_min_dmg:get()
    local min_hc = cfg.dormant_hc:get()

    entity.get_players(true, true, function(enemy)
        local net_state = enemy:get_network_state()
        if net_state == 0 or net_state == 5 then return end
        local bbox = enemy:get_bbox()
        if not bbox then return end
        local alpha = bbox.alpha
        if 4 - math.clamp(alpha, 0, 0.8) * 5 >= cfg.dormant_timeout:get() then return end
        local origin = enemy:get_origin()
        origin.z = origin.z + math.random(32, 50)
        local trace = utils.trace_bullet(lp, eye_pos, origin, function(e) return e == enemy end)
        local health = state.dormant_target_data[enemy:get_index()] or enemy.m_iHealth
        if not trace or trace[1] < min_dmg then return end
        local direction = eye_pos:to(origin)
        local dir_norm = direction:normalized()
        local spread = weapon:get_spread()
        local inacc = weapon:get_inaccuracy()
        local hits = 0
        for _ = 1, 128 do
            local angle = vector(
                math.cos(math.random() * math.pi*2) * math.random() * spread + math.cos(math.random() * math.pi*2) * math.random() * inacc,
                math.sin(math.random() * math.pi*2) * math.random() * spread + math.sin(math.random() * math.pi*2) * math.random() * inacc
            )
            local dir = vector(dir_norm.x + angle.x, dir_norm.y + angle.y, dir_norm.z)
            if origin:dist_to_ray(eye_pos, dir) <= 16 then hits = hits + 1 end
        end
        local hitchance = hits / 128 * 100
        if hitchance < min_hc then return end
        table.insert(candidates, {
            entity = enemy,
            health = health,
            damage = trace[1],
            origin = origin,
            hitchance = hitchance,
            eye_pos = eye_pos
        })
    end)

    if #candidates == 0 then return end

    local target = nil
    local sel = cfg.dormant_target_sel:get()
    if sel == "Best Hit Chance" then
        table.sort(candidates, function(a,b) return a.hitchance > b.hitchance end)
        target = candidates[1]
    elseif sel == "Best Damage" then
        table.sort(candidates, function(a,b) return a.damage > b.damage end)
        target = candidates[1]
    elseif sel == "Closest to Crosshair" then
        local cross = render.camera_angles()
        local cross_dir = vector():angles(cross)
        local min_dist = math.huge
        for _, c in ipairs(candidates) do
            local dir = eye_pos:to(c.origin)
            local dist = math.abs(math.normalize_yaw(dir:angles().y - cross.y))
            if dist < min_dist then
                min_dist = dist
                target = c
            end
        end
    elseif sel == "Lowest Health" then
        table.sort(candidates, function(a,b) return a.health < b.health end)
        target = candidates[1]
    end

    if target then
        if cfg.dormant_auto_stop:get() then
            cmd.block_movement = cfg.dormant_stop_mode:get("Full Stop") and 2 or 1
        end
        if cfg.dormant_auto_scope:get() and weapon_info.weapon_type == 5 and not lp.m_bIsScoped then
            cmd.in_attack2 = true
            return
        end
        cmd.in_attack = true
        cmd.view_angles = (target.eye_pos:to(target.origin)):angles()
    end
end

    local in_air = bit.band(lp.m_fFlags, 1) == 0

    local view_angles = vector():angles(cmd.view_angles)
    local vel = lp.m_vecVelocity
    local dir = (view_angles * speed + vel * -1.25) / speed
    dir.z = 0
    dir:normalize()
    if cmd.choked_commands == 0 then
        state.super_toss_last_dir = dir
    else
        if state.super_toss_last_dir then
            dir = (dir + state.super_toss_last_dir) * 0.5
        end
    end
    if can_throw then
        cmd.view_angles.y = dir:angles().y
    end
end

-- 9.15 R8 HELPER
function funcs.update_r8_data()
    if not cfg.advanced_rage_enable or not cfg.advanced_rage_enable:get() then return end
    if not cfg.r8_enable:get() then return end
    local lp = entity.get_local_player()
    if not lp or not lp:is_alive() then return end
    local weapon = lp:get_player_weapon()
    if not weapon or weapon:get_weapon_index() ~= 64 then return end
    local weapon_info = weapon:get_weapon_info()
    if not weapon_info then return end
    local damage_base = weapon_info.damage * 1.25
    local armor_ratio = weapon_info.armor_ratio
    local range_mod = weapon_info.range_modifier
    local eye_pos = lp:get_eye_position()
    for _, enemy in ipairs(entity.get_players(true, false)) do
        if enemy:is_alive() and not enemy:is_dormant() then
            local chest = enemy:get_hitbox_position(2)
            local trace = utils.trace_line(eye_pos, chest, function(e) return e == enemy end, 1174421507, 2)
            local dist = trace.start_pos:dist(trace.end_pos)
            local damage = damage_base * range_mod ^ (dist * 0.002)
            if enemy.m_ArmorValue > 0 then
                damage = damage * (armor_ratio * 0.5)
            end
            state.r8_data[enemy:get_index()] = {health = enemy.m_iHealth, damage = damage}
        end
    end
end

function funcs.draw_r8_lines()
    if not cfg.advanced_rage_enable or not cfg.advanced_rage_enable:get() then return end
    if not cfg.r8_enable:get() or not cfg.r8_snap_lines:get() then return end
    local lp = entity.get_local_player()
    if not lp or not lp:is_alive() then return end
    local weapon = lp:get_player_weapon()
    if not weapon or weapon:get_weapon_index() ~= 64 then return end
    local eye_pos = lp:get_eye_position()
    local screen_center = render.screen_size() * 0.5
    local closest_dist = math.huge
    local closest_pos = nil
    local closest_health, closest_damage = nil, nil
    for _, enemy in ipairs(entity.get_players(true, false)) do
        if enemy:is_alive() and not enemy:is_dormant() then
            local data = state.r8_data[enemy:get_index()]
            if data then
                local health, damage = data.health, data.damage
                if health - damage <= cfg.r8_range:get() then
                    local chest = enemy:get_hitbox_position(2)
                    local screen_pos = chest:to_screen()
                    if screen_pos then
                        local dist = eye_pos:dist(chest)
                        if dist < closest_dist then
                            closest_dist = dist
                            closest_pos = screen_pos
                            closest_health, closest_damage = health, damage
                        end
                    end
                end
            end
        end
    end
    if closest_pos then
        local alpha = math.map(math.max(closest_health - cfg.r8_range:get(), math.min(closest_health, closest_damage)), closest_health - cfg.r8_range:get(), closest_health, 0, 1)
        local col = cfg.r8_snap_color:get():clone()
        col.a = col.a * alpha
        render.line(screen_center, closest_pos, col)
    end
end

-- 9.16 BETTER SCOPE OVERLAY
function funcs.draw_better_scope()
    if not cfg.dora_visuals_enable or not cfg.dora_visuals_enable:get() then return end
    if not cfg.better_scope:get() then return end
    local lp = entity.get_local_player()
    if not lp or not lp:is_alive() then return end
    local weapon = lp:get_player_weapon()
    if not weapon then return end
    local is_scoped = lp.m_bIsScoped
    if not is_scoped then return end

    local center = screen * 0.5

    if is_scoped then
        state.scope_alpha = math.min(1, state.scope_alpha + globals.exactframetime * 3)
    else
        state.scope_alpha = math.max(0, state.scope_alpha - globals.exactframetime * 10)
    end
    local alpha = state.scope_alpha

    local line = cfg.scope_line:get() / 100
    local gap = cfg.scope_gap:get() / 100
    local aspect = cfg.scope_aspect:get() / 100
    local line_w = math.ceil(screen.y * 0.5 * line)
    local gap_w = math.ceil(screen.y * 0.5 * gap)
    if cfg.scope_spread:get() ~= "Disabled" then
        local inacc = cfg.scope_spread:get() == "Inaccuracy" and weapon:get_inaccuracy() or weapon.m_fAccuracyPenalty
        local spread = weapon:get_spread()
        if inacc and spread then
            local extra = math.ceil((inacc + spread) * 320 / math.tan(math.rad(cfg.refs.fov or 90) * 0.5))
            line_w = line_w + extra
            gap_w = gap_w + extra
        end

    if not cfg.scope_exclude:get("Top") then
        render.gradient(vector(center.x, center.y - gap_w), vector(center.x + 1, center.y - line_w), col, col:with_alpha(0), col, col:with_alpha(0))
    end
    if not cfg.scope_exclude:get("Bottom") then
        render.gradient(vector(center.x, center.y + gap_w), vector(center.x + 1, center.y + line_w), col, col:with_alpha(0), col, col:with_alpha(0))
    end
    if not cfg.scope_exclude:get("Left") then
        render.gradient(vector(center.x - gap_w, center.y), vector(center.x - line_w, center.y + 1), col, col:with_alpha(0), col, col:with_alpha(0))
    end
    if not cfg.scope_exclude:get("Right") then

-- 9.17 OFFSCREEN ESP
function funcs.draw_offscreen_esp()
    if not cfg.dora_visuals_enable or not cfg.dora_visuals_enable:get() then return end
    if not cfg.offscreen_esp:get() then return end
    local lp = entity.get_local_player()
    if not lp then return end
    local observer = lp.m_hObserverTarget
    if not lp:is_alive() and observer then
        lp = observer
    end
    local screen = render.screen_size()
    local center = screen * 0.5
    local cam_pos = render.camera_position()
    local cam_angle = render.camera_angles().y
    local radius = screen.y * 0.5 * cfg.offscreen_radius:get() / 100
    local size = screen.y * 0.125 * cfg.offscreen_size:get() / 100
    state.offscreen_pulse = state.offscreen_pulse + (state.offscreen_dir and globals.exactframetime * 1.25 or -globals.exactframetime * 1.25)
    if state.offscreen_pulse >= 1 or state.offscreen_pulse <= 0 then state.offscreen_dir = not state.offscreen_dir end
    state.offscreen_pulse = math.clamp(state.offscreen_pulse, 0, 1)

    for _, enemy in ipairs(entity.get_players(false, false)) do
        if enemy:is_alive() and enemy.m_iTeamNum ~= lp.m_iTeamNum then
            local pos = enemy:get_origin()
            pos.z = pos.z + enemy.m_vecMaxs.z * 0.5
            local screen_pos = render.world_to_screen(pos)
            if not screen_pos or screen_pos.x <= 0 or screen_pos.x >= screen.x or screen_pos.y <= 0 or screen_pos.y >= screen.y then
                local offscreen = render.get_offscreen(pos, 1, true)
                local angle
                if cfg.offscreen_alt:get() then
                    angle = (center:to(offscreen)):angles().y + 90
                else
                    angle = math.normalize_yaw(cam_angle - cam_pos:to(pos):angles().y)
                    angle = math.max(-90, math.min(90, math.normalize_yaw(angle + 180)))
                    angle = math.normalize_yaw(angle - 180)
                end
                angle = math.normalize_yaw(angle - 90)
                local rad = math.rad(angle)
                local pos2 = vector(
                    center.x + math.cos(rad) * radius * (screen.x/screen.y),
                    center.y + math.sin(rad) * radius
                )
                local tip = vector(
                    pos2.x + math.cos(rad + math.pi) * size,
                    pos2.y + math.sin(rad + math.pi) * size
                )
                local left = vector(
                    tip.x + math.cos(rad + math.pi/2) * (size * 0.5),
                    tip.y + math.sin(rad + math.pi/2) * (size * 0.5)
                )
                local right = vector(
                    tip.x + math.cos(rad - math.pi/2) * (size * 0.5),
                    tip.y + math.sin(rad - math.pi/2) * (size * 0.5)
                )
                local col = cfg.offscreen_color:get():clone()
                col.a = col.a * state.offscreen_pulse
                render.poly(col, left, pos2, right)
            end
        end
    end
end

-- 9.21 ХИТЛОГ (отрисовка)
function funcs.draw_hitlogs()
    if not cfg.dora_visuals_enable or not cfg.dora_visuals_enable:get() then return end
    if not cfg.hitlog_enable:get() then return end
    local screen = render.screen_size()
    for i = #state.hitlog_items, 1, -1 do
        local item = state.hitlog_items[i]
        if globals.tickcount >= item[2] then
            if item[3] > 0 then item[3] = item[3] - 20
            else table.remove(state.hitlog_items, i) end
        end
    end
    if #state.hitlog_items > 6 then table.remove(state.hitlog_items, 1) end
    if not globals.is_connected then state.hitlog_items = {} end

    for i, item in ipairs(state.hitlog_items) do
        local txt = item[1]
        local sz = render.measure_text(1, nil, txt).x
        local sz2 = render.measure_text(1, nil, "[dora.lua] ").x
        if item[3] < 255 then item[3] = item[3] + 10 end
        if cfg.hitlog_position:get("Screen middle") then
            local bx = screen.x/2 - sz/2
            local by = screen.y/1.5 + 15*i
            local off = item[3] / 35
            local offy = item[3] / 45
            local pref = "[dora.lua]"
            local anim = cfg.hitlog_animation:get()
            local clr = cfg.hitlog_color:get()
            if anim == 'None' then
                render.text(1, vector(bx+sz2, by), color(255,255,255,item[3]), nil, txt)
                render.text(1, vector(bx, by), color(clr.r,clr.g,clr.b,item[3]), nil, pref)
            elseif anim == 'X +' then
                render.text(1, vector(bx+off+sz2, by), color(255,255,255,item[3]), nil, txt)
                render.text(1, vector(bx+off, by), color(clr.r,clr.g,clr.b,item[3]), nil, pref)
            elseif anim == 'X -' then
                render.text(1, vector(bx-off+sz2, by), color(255,255,255,item[3]), nil, txt)
                render.text(1, vector(bx-off, by), color(clr.r,clr.g,clr.b,item[3]), nil, pref)
            elseif anim == 'Y +' then
                render.text(1, vector(bx+sz2, by+offy), color(255,255,255,item[3]), nil, txt)
                render.text(1, vector(bx, by+offy), color(clr.r,clr.g,clr.b,item[3]), nil, pref)
            elseif anim == 'Y -' then
                render.text(1, vector(bx+sz2, by-offy), color(255,255,255,item[3]), nil, txt)
                render.text(1, vector(bx, by-offy), color(clr.r,clr.g,clr.b,item[3]), nil, pref)
            end
        end
    end
end

-- 9.22 ГЛАВНЫЙ ОБРАБОТЧИК CREATEMOVE
function funcs.on_createmove(cmd)
    local lp = entity.get_local_player()
    if not lp or not lp:is_alive() then return end
    local weapon = lp:get_player_weapon()
    if not weapon then return end

    local override = utils.safe_override or function() end

    -- Jumpscout fix (зависит от мастер-переключателя Misc)
    if cfg.misc_enable and cfg.misc_enable:get() and cfg.jumpscout_enable and cfg.jumpscout_enable:get() then
        local speed = lp.m_vecVelocity:length2d()
        override(ui_refs.refs.air_strafe, speed > 15)
    else
        override(ui_refs.refs.air_strafe)
    end

    -- Flicks
    local flick_angle_val = funcs.update_flick(cmd, lp)
    if flick_angle_val ~= 0 then
        override(ui_refs.refs.aa_yaw_offset, flick_angle_val)
    else
        override(ui_refs.refs.aa_yaw_offset)
    end

    -- Anti-Aim по состояниям (Better AA)
    local state_name = funcs.get_aa_state(lp, weapon, cmd)
    funcs.apply_aa_state(state_name)

    -- Chimera функции (зависят от мастер-переключателя Advanced Rage)
    if cfg.advanced_rage_enable and cfg.advanced_rage_enable:get() then
        if cfg.chimera_general_enable and cfg.chimera_general_enable:get() then
            funcs.ai_peek_on_createmove(cmd)
            funcs.dormant_aimbot_on_createmove(cmd)
            funcs.magic_key_on_createmove()
            funcs.ideal_tick_on_createmove(cmd, lp, weapon)
            funcs.super_toss_on_createmove(cmd)
            funcs.fast_ladder_on_createmove(cmd)
            funcs.unlock_fake_latency()
        end
    end
end

function funcs.on_player_death(e)
    local lp = entity.get_local_player()
    local attacker = entity.get(e.attacker, true)
    if attacker == lp then
        state.round_kills = state.round_kills + 1
        if cfg.misc_enable and cfg.misc_enable:get() and cfg.trash_talk_enable:get() and attacker ~= entity.get(e.userid, true) then
            local victim = entity.get(e.userid, true)
            if victim then
                local victim_name = victim:get_name()
                local mode = cfg.trash_talk_mode:get()
                local phrases = mode == "Quotes list" and cfg.preset_phrases or cfg.custom_phrases_list:get()
                if #phrases > 0 then
                    local msg = phrases[math.random(1, #phrases)]
                    msg = msg:gsub("%%name%%", victim_name)
                    state.chat_queue.add(msg, cfg.trash_talk_delay:get())
                end
            end
        end
    end
end

function funcs.on_aim_ack(e)
    if not cfg.dora_visuals_enable or not cfg.dora_visuals_enable:get() then return end
    if not cfg.hitlog_enable:get() or not cfg.hitlog_log_aimbot:get() then return end
    local target = entity.get(e.target)
    if not target then return end
    
    local health = target["m_iHealth"]
    local msg, is_miss
    local hitgroup_names = {
        [0] = 'generic', [1] = 'head', [2] = 'chest', [3] = 'stomach', 
        [4] = 'left arm', [5] = 'right arm', [6] = 'left leg', 
        [7] = 'right leg', [8] = 'neck', [9] = 'generic', [10] = 'gear'
    }

    if e.state == nil then
        msg = ("Registered shot at %s's %s(%s%%) for %s (aimed: %s for %s, health: %s) backtrack: %s"):format(
            target:get_name(), 
            hitgroup_names[e.hitgroup] or "unknown", 
            math.floor(e.hitchance or 0), 
            e.damage,
            hitgroup_names[e.wanted_hitgroup] or "unknown", 
            e.wanted_damage, 
            health, 
            e.backtrack
        )
        is_miss = false
    else
        local reason_str = e.state
        if reason_str == "correction" then reason_str = "resolver"
        elseif reason_str == "lagcomp failure" then reason_str = "fake lag correction" end
        
        msg = ("Missed %s's %s (dmg:%s, %s%%) due to %s | backtrack: %s"):format(
            target:get_name(), 
            hitgroup_names[e.wanted_hitgroup] or "unknown", 
            e.wanted_damage, 
            math.floor(e.hitchance or 0), 
            reason_str, 
            e.backtrack
        )
        is_miss = true
    end
    table.insert(state.hitlog_items, {msg, globals.tickcount + 250, 0})
end

-- 12.3 Функция определения ID состояния (как в Chimera)
function funcs.get_aa_state_id(cmd, lp, weapon)
    local vel = lp.m_vecVelocity:length2d()
    local duck = lp.m_flDuckAmount > 0.99
    local on_ground = bit.band(lp.m_fFlags, 1) == 1
    local slowwalk = cfg.refs.aa_slow_walk:get_override() or cfg.refs.aa_slow_walk:get()
    local fd = cfg.refs.aa_fake_duck:get_override() or cfg.refs.aa_fake_duck:get()
    local in_jump = cmd.in_jump

    local state_id = 2 -- Standing
    if vel > 2 then
        state_id = 3 -- Moving
        if slowwalk then state_id = 4 end -- SlowWalking
    end
    if duck or fd then
        state_id = 5 -- Crouching
    end
    if not on_ground then
        state_id = duck and 7 or 6 -- InAirCrouch or InAir
    end
    if weapon and weapon:get_classid() == 34 then
        state_id = 8 -- InUse (C4)
    end
    if cmd.in_use then
        state_id = 8
    end
    local dt = cfg.refs.dt:get_override() or cfg.refs.dt:get()
    local hs = cfg.refs.hs:get_override() or cfg.refs.hs:get()
    if not dt and not hs then
        state_id = 9
    end
    return state_id
end

-- 12.4 Функция применения настроек для текущего состояния (Better Angles)
-- ======================================================================
function funcs.apply_aa_state_better_angles(state_name)
    -- Проверка мастер-переключателя Better Angles
    if not cfg.better_angles_enable or not cfg.better_angles_enable:get() then return end
    local cfg_state = state.aa_state_groups[state_name]
    if not cfg_state or not cfg_state.enabled:get() then return end

    local r = ui_refs.refs

    local pitch_val = cfg_state.pitch:get()
    if pitch_val == "Random" then
        local rand = math.random(1, 3)
        pitch_val = rand == 1 and "Down" or rand == 2 and "Up" or "Zero"
    end

    if pitch_val ~= "Default" then
        utils.safe_override(r.aa_pitch, pitch_val)
    else
        utils.safe_override(r.aa_pitch)
    end

    local yaw_val = cfg_state.yaw:get()
    if yaw_val ~= "Default" then
        utils.safe_override(r.aa_yaw, yaw_val)
    else
        utils.safe_override(r.aa_yaw)
    end

    local inv = cfg_state.inverter:get()
    local yaw_off = inv and cfg_state.yaw_offset_left:get() or cfg_state.yaw_offset_right:get()
    utils.safe_override(r.aa_yaw_offset, yaw_off)

    local mod = cfg_state.yaw_modifier:get()
    local mod_deg = cfg_state.modifier_degree:get()
    
    if mod == "Center" then
        rage.antiaim:override_hidden_yaw_offset(mod_deg)
    elseif mod == "Offset" then
        rage.antiaim:override_hidden_yaw_offset(yaw_off + mod_deg)
    elseif mod == "Random" then
        rage.antiaim:override_hidden_yaw_offset(yaw_off + math.random(-mod_deg, mod_deg))
    else
        rage.antiaim:override_hidden_yaw_offset(0)
    end

    local body_yaw_val = cfg_state.body_yaw:get()
    if body_yaw_val ~= "Default" then
        utils.safe_override(r.aa_body_yaw, body_yaw_val)
    else
        utils.safe_override(r.aa_body_yaw)
    end

    utils.safe_override(r.aa_inverter, inv)
    utils.safe_override(r.aa_left_limit, cfg_state.left_limit:get())
    utils.safe_override(r.aa_right_limit, cfg_state.right_limit:get())
end

-- ФИНАЛЬНЫЙ ОБРАБОТЧИК CREATEMOVE (с Better AA и Better Angles)
-- ======================================================================
function funcs.on_createmove(cmd)
    local lp = entity.get_local_player()
    if not lp or not lp:is_alive() then return end
    local weapon = lp:get_player_weapon()
    if not weapon then return end

    local override = utils.safe_override or function() end

    -- Jumpscout fix
    if cfg.misc_enable and cfg.misc_enable:get() and cfg.jumpscout_enable and cfg.jumpscout_enable:get() then
        local speed = lp.m_vecVelocity:length2d()
        override(ui_refs.refs.air_strafe, speed > 15)
    else
        override(ui_refs.refs.air_strafe)
    end

    -- Flicks
    local flick_angle_val = funcs.update_flick(cmd, lp)
    if flick_angle_val ~= 0 then
        override(ui_refs.refs.aa_yaw_offset, flick_angle_val)
    else
        override(ui_refs.refs.aa_yaw_offset)
    end

    -- Anti-Aim по состояниям (Better AA)
    local state_name = funcs.get_aa_state(lp, weapon, cmd)
    funcs.apply_aa_state(state_name)

    -- Расширенный Anti-Aim по состояниям (Better Angles)
    local state_id = funcs.get_aa_state_id(cmd, lp, weapon)
    local angles_state_name = state.aa_state_names[state_id] or "Default"
    funcs.apply_aa_state_better_angles(angles_state_name)

    -- Chimera функции
    if cfg.advanced_rage_enable and cfg.advanced_rage_enable:get() then
        if cfg.chimera_general_enable and cfg.chimera_general_enable:get() then
            funcs.ai_peek_on_createmove(cmd)
            funcs.dormant_aimbot_on_createmove(cmd)
            funcs.magic_key_on_createmove()
            funcs.ideal_tick_on_createmove(cmd, lp, weapon)
            funcs.super_toss_on_createmove(cmd)
            funcs.fast_ladder_on_createmove(cmd)
            funcs.unlock_fake_latency()
        end
    end
end

-- 13.1 Safe Head (Chimera)
function funcs.safe_head_on_createmove(cmd, lp, weapon)
    if not cfg.advanced_rage_enable or not cfg.advanced_rage_enable:get() then return end
    if not cfg.aa_extras.safe_head:get() then return end
    local enemy = entity.get_threat()
    if not enemy then return end
    local eye_pos = lp:get_eye_position()
    local enemy_origin = enemy:get_origin()
    local dist = eye_pos:dist2dsqr(enemy_origin) / 10000
    enemy_origin.z = enemy_origin.z + 64
    local height_diff = eye_pos.z - enemy_origin.z
    local state_id = funcs.get_aa_state_id(cmd, lp, weapon)
    local safe_conditions = {
        Standing = state_id == 2 and height_diff > 22,
        Crouching = state_id == 5 and height_diff > 22,
        AirCrouching = state_id == 7 and height_diff > 65,
        Distance = (state_id == 2 and dist > 100 and height_diff >= 0) or (state_id == 5 and dist > 100 and height_diff >= -18),
        ["Бизончик)"] = state_id == 2 and weapon and weapon:get_classid() == 235 and height_diff >= -18
    }
    local should_activate = false
    for _, cond in ipairs(cfg.aa_extras.safe_head:get()) do
        if safe_conditions[cond] then should_activate = true; break end
    end
    if should_activate then
        rage.antiaim:override_hidden_pitch(89)
        utils.safe_override(cfg.refs.aa_pitch, "Down")
    end
end

-- 13.2 Flick Exploit (Chimera)
function funcs.flick_exploit_on_createmove(cmd, lp, weapon)
    if not cfg.advanced_rage_enable or not cfg.advanced_rage_enable:get() then return end
    if not cfg.aa_extras.flick_exploit:get() then return end
    local state_id = funcs.get_aa_state_id(cmd, lp, weapon)
    local conditions = {}
    for _, cond in ipairs(cfg.aa_extras.flick_exploit:get()) do
        if cond == "Standing" and state_id == 2 then conditions[#conditions+1] = true end
        if cond == "Moving" and state_id == 3 then conditions[#conditions+1] = true end
        if cond == "Slow Walking" and state_id == 4 then conditions[#conditions+1] = true end
        if cond == "Crouching" and state_id == 5 then conditions[#conditions+1] = true end
        if cond == "In-Air" and state_id == 6 then conditions[#conditions+1] = true end
        if cond == "In-Air Crouching" and state_id == 7 then conditions[#conditions+1] = true end
    end
    if #conditions > 0 then
        local side = math.random(0,1) == 0 and -180 or 180
        if cfg.aa_extras.flick_exploit_alt:get() then
            rage.antiaim:override_hidden_yaw_offset(side)
        else
            rage.antiaim:override_hidden_yaw_offset(180)
        end
        rage.antiaim:override_hidden_pitch(89)
    end
end

-- 13.4 Force Break LC (Chimera)
function funcs.force_break_lc_on_createmove(cmd, lp, weapon)
    if not cfg.advanced_rage_enable or not cfg.advanced_rage_enable:get() then return end
    local lc_opts = cfg.aa_extras.force_break_lc:get()
    if #lc_opts == 0 then return end
    local state_id = funcs.get_aa_state_id(cmd, lp, weapon)
    local conditions = {}
    for _, cond in ipairs(lc_opts) do
        if cond == "Standing" and state_id == 2 then conditions[#conditions+1] = true end
        if cond == "Moving" and state_id == 3 then conditions[#conditions+1] = true end
        if cond == "Slow Walking" and state_id == 4 then conditions[#conditions+1] = true end
        if cond == "Crouching" and state_id == 5 then conditions[#conditions+1] = true end
        if cond == "In Air" and state_id == 6 then conditions[#conditions+1] = true end
        if cond == "Safe Head" and (cfg.aa_extras.safe_head:get() and state_id == 2) then conditions[#conditions+1] = true end
        if cond == "Manual Yaw Base" and cfg.aa_extras.manual_yaw_enabled then conditions[#conditions+1] = true end
    end
    if #conditions > 0 then
        cfg.refs.dt_lag_options:override("Always On")
        cfg.refs.hs_options:override("Break LC")
    else
        cfg.refs.dt_lag_options:override()
        cfg.refs.hs_options:override()
    end
end

-- 13.5 Disable Fake Lag (Chimera)
function funcs.disable_fake_lag_on_createmove(cmd, lp, weapon)
    if not cfg.advanced_rage_enable or not cfg.advanced_rage_enable:get() then return end
    local disable_opts = cfg.aa_extras.disable_fake_lag:get()
    if #disable_opts == 0 then return end
    local disable = false
    for _, opt in ipairs(disable_opts) do
        if opt == "Double Tap" and (cfg.refs.dt:get_override() or cfg.refs.dt:get()) then
            disable = true
        elseif opt == "Hide Shots" and (cfg.refs.hs:get_override() or cfg.refs.hs:get()) then
            disable = true
        elseif opt == "Standing" and funcs.get_aa_state_id(cmd, lp, weapon) == 2 then
            disable = true
        elseif opt == "DT Heavy Pistols" and (cfg.refs.dt:get_override() or cfg.refs.dt:get()) and (weapon:get_weapon_index() == 1 or weapon:get_weapon_index() == 64) then
            disable = true
        elseif opt == "DT Knife" and (cfg.refs.dt:get_override() or cfg.refs.dt:get()) and weapon:get_classid() == 107 then
            disable = true
        elseif opt == "DT High Firerate" and (cfg.refs.dt:get_override() or cfg.refs.dt:get()) and (weapon:get_weapon_info().weapon_type == "SMG" or weapon:get_weapon_info().weapon_type == "Pistol") then
            disable = true
        end
    end
    utils.safe_override(cfg.refs.aa_fakelag_enable, not disable)
end

-- 13.6 Автоматический телепорт (Chimera)
function funcs.automatic_teleport_on_createmove(cmd, lp, weapon)
    if not cfg.advanced_rage_enable or not cfg.advanced_rage_enable:get() then return end
    if not cfg.aa_extras.automatic_teleport:get() then return end
    local weapon_type = weapon:get_weapon_info().weapon_type
    if not cfg.aa_extras.teleport_weapons:get(weapon_type) then return end
    if cfg.aa_extras.teleport_air_only:get() and bit.band(lp.m_fFlags, 1) == 1 then return end
    local enemy = entity.get_threat(true)
    if enemy then
        if not state.teleport_charging then
            state.teleport_charge_tick = globals.tickcount + cfg.aa_extras.teleport_delay:get()
            state.teleport_charging = true
        end
        if globals.tickcount >= state.teleport_charge_tick then
            rage.exploit:force_teleport()
            state.teleport_charging = false
        end
    else
        state.teleport_charging = false
    end
end

-- 13.8 Aerobic Lag Exploit
function funcs.aerobic_lag_exploit_on_createmove(cmd, lp, weapon)
    if not cfg.advanced_rage_enable or not cfg.advanced_rage_enable:get() then return end
    if not cfg.aa_extras.aerobic_lag_exploit:get() then return end
    if cfg.aa_extras.aerobic_while_visible:get() and not entity.get_threat(true) then return end
    if (lp.m_vecVelocity * globals.tickinterval * 14):lengthsqr() <= 4096 then return end
    if bit.band(lp.m_fFlags, 1) == 1 then return end
    rage.exploit:force_charge()
    local tick = cmd.command_number % 6
    if tick == 0 then
        rage.exploit:force_teleport()
    elseif tick == 1 or tick == 5 then
        cmd.force_defensive = true
    end
end

-- 13.11 Unlock FD Speed
function funcs.unlock_fd_speed_on_createmove(cmd, lp, weapon)
    if not cfg.advanced_rage_enable or not cfg.advanced_rage_enable:get() then return end
    if not cfg.aa_extras.unlock_fd_speed:get() then return end
    if cfg.refs.aa_fake_duck:get_override() and not cmd.in_speed then
        local vel = math.sqrt(cmd.forwardmove^2 + cmd.sidemove^2)
        if vel > 5 and vel < 100 then
            local factor = 110 / vel
            cmd.forwardmove = cmd.forwardmove * factor
            cmd.sidemove = cmd.sidemove * factor
        end
    end
end

-- Функция предсказания урона (Damage Prediction)
function funcs.get_predicted_damage(ent)
    local lp = entity.get_local_player()
    if not lp or not lp:is_alive() or not ent then return 0 end
    
    local weapon_ent = lp:get_player_weapon()
    if not weapon_ent then return 0 end
    local weapon_info = weapon_ent:get_weapon_info()
    if not weapon_info then return 0 end

    local target_pos = ent:get_hitbox_pos(1)
    local lp_pos = lp:get_eye_position()
    local distance = lp_pos:dist_to(target_pos)

    local base_dmg = weapon_info.m_iDamage
    local range_modifier = weapon_info.m_flRangeModifier
    
    local dmg_after_range = base_dmg * math.pow(range_modifier, distance * 0.002)
    
    local hitbox_sel = cfg.dmg_hitbox_priority:get()
    local multiplier = 1.0
    if hitbox_sel == "Head" then multiplier = 4.0 
    elseif hitbox_sel == "Stomach" then multiplier = 1.25 
    elseif hitbox_sel == "Legs" then multiplier = 0.75 end

    return math.floor(dmg_after_range * multiplier)
end

-- 7.3 Рендер предсказания урона
function funcs.draw_damage_prediction()
    if not cfg.dmg_predict_enable or not cfg.dmg_predict_enable:get() then return end
    
    local target = rage.exploit:get_target() or rage.antiaim:get_target()
    
    if target and target:is_alive() then
        local p_dmg = funcs.get_predicted_damage(target)
        if p_dmg and p_dmg > 0 then
            local hitbox_pos = target:get_hitbox_pos(1)
            local screen_pos = render.world_to_screen(hitbox_pos)
            
            if screen_pos then
                local color_text = color(255, 255, 255, 255)
                if p_dmg >= target.m_iHealth then
                    color_text = color(255, 0, 0, 255)
                end
                
                render.text(1, screen_pos, color_text, "c", "PRED: " .. p_dmg)
            end
        end
    end
end
